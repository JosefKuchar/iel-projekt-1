    ;
  \end{circuitikz}

  \begin{equation*}
    \begin{align}
      R_A &= \frac{R_1 \cdot R_2}{R_1 + R_2 + R_3} \\
      R_B &= \frac{R_1 \cdot R_3}{R_1 + R_2 + R_3} \\
      R_C &= \frac{R_2 \cdo R_3}{R_1 + R_2 + R_3}
    \end{align}
  \end{equation*}
